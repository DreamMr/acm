#include <iostream>

using namespace std;
int a[100001]={0};
int main()
{
    int n=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        int t=0,s=0,sum=0,res=0,ss=0;
        cin>>t>>ss;
        res=t+1;
        int j=0;
        for(;j<t;j++)
        {
            cin>>a[j];
        }
        j=0;
        while(1)
        {

            while(j<t && sum<ss)
            {
                sum+=a[j++];
            }
            if(sum<ss){break;}
            res=min(res,j-s);
            sum-=a[s++];
        }
        if(res>t)
        {
            cout<<0<<endl;
        }
        else
        {
            cout<<res<<endl;
        }
    }
    return 0;
}
