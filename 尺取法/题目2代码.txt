#include<set>
#include<map>
#include<stdio.h>
using namespace std;
int main()
{
    int n=0;
   /* while(cin>>n)
    {*/
    //cin>>n;
    scanf("%d",&n);
        int *a=new int [n];
        set<int>s;
        int res=n+1;
        int num=0;
        for(int i=0;i<n;i++)
        {
            //cin>>a[i];
            scanf("%d",&a[i]);
            s.insert(a[i]);
        }
        int t=s.size();
        map<int,int>m;
        int i=0,j=0;
        while(1)
        {
            while(i<n && num<t)
            {
                if(m[a[i++]]++==0)
                {
                    num++;
                }
            }
            if(num<t){break;}
            res=min(res,i-j);
            if(--m[a[j++]]==0)
            {
                num--;
            }
        }
        //cout<<res<<endl;
        printf("%d\n",res);
}
