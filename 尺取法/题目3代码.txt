#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[10001]={0};
int b[10001]={0};
int t=0;
void isprime()
{
    a[0]=1;
    a[1]=1;
    for(int i=2;i*i<10001;i++)
    {
        if(!a[i])
        {
            int j=2*i;
            while(j<10001)
            {
                a[j]=1;
                j+=i;
            }
        }
    }

    for(int i=2;i<10001;i++)
    {
        if(!a[i])
        {
           b[t++]=i;
        }
    }
}
int main()
{
    isprime();
    int n=0;
    while(scanf("%d",&n)!=EOF)
    {
        if(!n){break;}
        int res=0;
        int sum=0;
        int s=0,e=0;
        while(b[e]<n && s<t)
        {
        while(e<t && sum<n)
        {
            sum+=b[e++];
        }
         if(sum==n)res++;
         sum-=b[s++];
        }
        if(!a[n])res++;
        printf("%d\n",res);
    }
    return 0;
}
